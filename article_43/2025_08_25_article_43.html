Este guia mostra, de forma simples, como transformar suas planilhas de vendas em um painel visual usando Python. Mesmo que você não seja analista de dados, vai conseguir montar uma visão clara para tomar decisões melhores.

História rápida: uma pequena loja de produtos naturais tinha 8 planilhas mensais diferentes. O dono só sabia “se vendeu bem” olhando o saldo bancário no fim da semana. Depois de organizar e visualizar dados em um dashboard simples (ticket médio, produtos que mais giram e horas de pico), ele ajustou estoque e reduziu perda de itens perecíveis em 22% em 2 meses.

Objetivo do Dashboard

  • Responder rápido: “Como estão as vendas do mês vs. mês passado?”
  • Encontrar produtos campeões e encalhados.
  • Prever necessidade de reposição básica.
  • Monitorar ticket médio e número de pedidos.

Componentes do Dashboard

  • Métricas Principais: Total de vendas, linhas de dados, última atualização
  • Tabelas Interativas: Vendas e estoque por produto em tempo real
  • Gráfico de Linha: Evolução temporal das vendas e estoque
  • Gráfico de Pizza: Distribuição de vendas por produto
  • Status de Conexão: Indicador visual de WebSocket/polling
  • Atualizações Automáticas: Dados atualizados a cada mudança no arquivo

Backend: FastAPI com WebSockets

from fastapi import FastAPI, WebSocket
from fastapi.responses import HTMLResponse
import pandas as pd
import json
from datetime import datetime

app = FastAPI()

class DataManager:
    def __init__(self):
        self.data = None
        self.connections = []
    
    def load_data(self, file_path):
        """Carrega dados do CSV e converte tipos numpy para JSON-safe."""
        try:
            df = pd.read_csv(file_path, parse_dates=['timestamp'])
            self.data = self.make_json_safe(df.to_dict('records'))
            return True
        except Exception as e:
            print(f"Erro ao carregar dados: {e}")
            return False
    
    def make_json_safe(self, data):
        """Converte tipos numpy/pandas para tipos nativos Python."""
        if isinstance(data, list):
            return [self.make_json_safe(item) for item in data]
        elif isinstance(data, dict):
            return {k: self.make_json_safe(v) for k, v in data.items()}
        elif hasattr(data, 'item'):  # numpy types
            return data.item()
        return data
    
    async def broadcast_update(self, message):
        """Envia atualizações para todos os clientes conectados."""
        if self.connections:
            disconnected = []
            for websocket in self.connections:
                try:
                    await websocket.send_json(message)
                except:
                    disconnected.append(websocket)
            
            # Remove conexões desconectadas
            for ws in disconnected:
                self.connections.remove(ws)

data_manager = DataManager()

@app.websocket("/ws")
async def websocket_endpoint(websocket: WebSocket):
    await websocket.accept()
    data_manager.connections.append(websocket)
    
    try:
        while True:
            await websocket.receive_text()
    except:
        data_manager.connections.remove(websocket)

@app.get("/api/data")
async def get_data():
    """Retorna dados agregados atuais."""
    if not data_manager.data:
        return {"error": "Dados não carregados"}
    
    df = pd.DataFrame(data_manager.data)
    return {
        "total_vendas": int(df["vendas"].sum()),
        "vendas_por_produto": df.groupby("produto")["vendas"].sum().to_dict(),
        "estoque_por_produto": df.groupby("produto")["estoque"].last().to_dict(),
        "linhas": len(df),
        "ultimo_timestamp": df["timestamp"].max().isoformat() if not df.empty else None,
        "atualizado_em": datetime.now().isoformat()
    }

Monitoramento de Arquivo em Tempo Real

from watchdog.observers import Observer
from watchdog.events import FileSystemEventHandler
import asyncio
import logging

class CSVHandler(FileSystemEventHandler):
    def __init__(self, data_manager, csv_path):
        self.data_manager = data_manager
        self.csv_path = csv_path
        self.logger = logging.getLogger(__name__)
    
    def on_modified(self, event):
        if event.src_path.endswith('.csv'):
            self.logger.info(f"Arquivo modificado: {event.src_path}")
            asyncio.create_task(self.process_update())
    
    async def process_update(self):
        """Processa atualização de dados e notifica clientes."""
        if self.data_manager.load_data(self.csv_path):
            # Calcula dados agregados
            data = await get_data()
            
            # Envia para todos os clientes conectados
            await self.data_manager.broadcast_update({
                "type": "data_update",
                "data": data,
                "timestamp": datetime.now().isoformat()
            })
            
            self.logger.info("Dados atualizados e enviados via WebSocket")

def setup_file_monitoring(data_manager, csv_path):
    """Configura monitoramento de arquivo com watchdog."""
    event_handler = CSVHandler(data_manager, csv_path)
    observer = Observer()
    observer.schedule(event_handler, path=str(csv_path.parent), recursive=False)
    observer.start()
    return observer

Frontend: Dashboard Interativo

class DashboardManager {
    constructor() {
        this.ws = null;
        this.isPolling = false;
        this.charts = {};
        this.reconnectInterval = 5000;
        this.initializeWebSocket();
        this.initializeCharts();
    }
    
    initializeWebSocket() {
        try {
            this.ws = new WebSocket(`ws://${window.location.host}/ws`);
            
            this.ws.onopen = () => {
                console.log('WebSocket conectado');
                this.updateConnectionStatus('WebSocket', 'conectado');
                this.stopPolling();
            };
            
            this.ws.onmessage = (event) => {
                const message = JSON.parse(event.data);
                if (message.type === 'data_update') {
                    this.updateDashboard(message.data);
                }
            };
            
            this.ws.onclose = () => {
                console.log('WebSocket desconectado - ativando polling');
                this.updateConnectionStatus('Polling', 'ativo');
                this.startPolling();
                setTimeout(() => this.initializeWebSocket(), this.reconnectInterval);
            };
        } catch (error) {
            console.log('WebSocket não disponível - usando polling');
            this.startPolling();
        }
    }
    
    startPolling() {
        if (this.isPolling) return;
        this.isPolling = true;
        
        const poll = async () => {
            try {
                const response = await fetch('/api/data');
                const data = await response.json();
                this.updateDashboard(data);
            } catch (error) {
                console.error('Erro no polling:', error);
            }
            
            if (this.isPolling) {
                setTimeout(poll, 3000);
            }
        };
        
        poll();
    }
    
    updateDashboard(data) {
        // Atualiza métricas principais
        document.getElementById('total-vendas').textContent = data.total_vendas;
        document.getElementById('total-linhas').textContent = data.linhas;
        document.getElementById('ultima-atualizacao').textContent = 
            new Date(data.ultimo_timestamp).toLocaleString();
        
        // Atualiza gráficos
        this.updateLineChart(data);
        this.updatePieChart(data);
        this.updateTables(data);
    }
    
    initializeCharts() {
        // Gráfico de linha para evolução temporal
        const lineCtx = document.getElementById('lineChart').getContext('2d');
        this.charts.line = new Chart(lineCtx, {
            type: 'line',
            data: { labels: [], datasets: [] },
            options: {
                responsive: true,
                plugins: { title: { display: true, text: 'Evolução de Vendas' } }
            }
        });
        
        // Gráfico de pizza para distribuição por produto
        const pieCtx = document.getElementById('pieChart').getContext('2d');
        this.charts.pie = new Chart(pieCtx, {
            type: 'pie',
            data: { labels: [], datasets: [] },
            options: {
                responsive: true,
                plugins: { title: { display: true, text: 'Vendas por Produto' } }
            }
        });
    }
}

// Inicializar dashboard
const dashboard = new DashboardManager();
